100void detect(Mat &image, dnn::Net &net, std::vector<Detection> &output, const std::vector<std::string> &className) {
101 Mat blob;
102
103 auto input_image = format_yolov5(image);
104 dnn::blobFromImage(input_image, blob, 1./255., Size(INPUT_WIDTH, INPUT_HEIGHT), Scalar(), true, false);
105 net.setInput(blob);
106 std::vector<Mat> outputs;
107 net.forward(outputs, net.getUnconnectedOutLayersNames());
108
109 float x_factor = input_image.cols / INPUT_WIDTH;
110 float y_factor = input_image.rows / INPUT_HEIGHT;
111
112 float *data = (float *)outputs[0].data;
113
114 const int dimensions = 6;
115 const int rows = 25200;
116
117 std::vector<int> class_ids;
118 std::vector<float> confidences;
119 std::vector<Rect> boxes;
120
121 for (int i = 0; i < rows; ++i) {
122
123 float confidence = data[4];
124 if (confidence >= CONFIDENCE_THRESHOLD) {
125
126 float * classes_scores = data + 5;
127 Mat scores(1, className.size(), CV_32FC1, classes_scores);
128 Point class_id;
129 double max_class_score;
130 minMaxLoc(scores, 0, &max_class_score, 0, &class_id);
131 if (max_class_score > SCORE_THRESHOLD) {
132
133 confidences.push_back(confidence);
134
135 class_ids.push_back(class_id.x);
136
137 float x = data[0];
138 float y = data[1];
139 float w = data[2];
140 float h = data[3];
141 int left = int((x - 0.5 * w) * x_factor);
142 int top = int((y - 0.5 * h) * y_factor);
143 int width = int(w * x_factor);
144 int height = int(h * y_factor);
145 boxes.push_back(Rect(left, top, width, height));
146 }
147
148 }
149 data +=6;
150
151 }
152 std::vector<int> nms_result;
153 dnn::NMSBoxes(boxes, confidences, SCORE_THRESHOLD, NMS_THRESHOLD, nms_result);
154 for (int i = 0; i < nms_result.size(); i++) {
155 int idx = nms_result[i];
156 Detection result;
157 result.class_id = class_ids[idx];
158 result.confidence = confidences[idx];
159 result.box = boxes[idx];
160 output.push_back(result);
161 }

163}
